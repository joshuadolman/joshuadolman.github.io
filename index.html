internal void
ParseDolIntoHTML(FileBuffer* dolFile, FileBuffer* htmlFile, char* fileName) {
  u64 offsetGain = 0;

  char* tempstr = "";
  strcat(((char*)htmlFile->memory.pointer + htmlFile->usedSize), tempstr);
  offsetGain+=strlen(tempstr);

  strcat(((char*)htmlFile->memory.pointer + htmlFile->usedSize), fileName);
  offsetGain+=strlen(fileName);

  tempstr = "";
  strcat(((char*)htmlFile->memory.pointer + htmlFile->usedSize), tempstr);
  offsetGain+=strlen(tempstr);

  htmlFile->atOffset+=offsetGain;
  htmlFile->usedSize+=offsetGain;

  while (dolFile->atOffset < dolFile->usedSize) {
    switch (*((char*)dolFile->memory.pointer + dolFile->atOffset)) {
      case '\t' :
      {
      char* tab = "  ";
      strcat(((char*)htmlFile->memory.pointer + htmlFile->atOffset), tab);
      htmlFile->atOffset+=strlen(tab);
      htmlFile->usedSize+=strlen(tab);
    } break;
  }
}
